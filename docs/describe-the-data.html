3 Describe the data

Describe the dataset

dat <- read_rds("data/players.rds")
titles <- read_rds("data/titles.rds")
dat <- left_join(titles, dat)
total <- dat %>% 
  group_by(Date) %>% 
  summarise(Players = sum(Players, na.rm = TRUE))
